```yaml architect.yml
static:
  website:
    directory: ./dist
    build:
      context: ./
      command: npm run build
      environment:
        API_ADDR: ${{ services.api.interfaces.main.url }}
    ingress:
      subdomain: app
      
services:
  api:
    build:
      context: ./
    interfaces:
      main:
        port: 8080
```
